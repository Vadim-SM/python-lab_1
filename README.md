# Програма для обчислення X в залежності від значень a та b

# Введення даних з клавіатури
a = int(input("Введіть число a (1 <= a <= 100): "))
b = int(input("Введіть число b (1 <= b <= 100): "))

# Перевірка правильності введених даних
if not (1 <= a <= 100 and 1 <= b <= 100):
    print("Помилка: числа повинні бути в межах від 1 до 100!")
else:
    # Обчислення X
    if a < b:
        X = (a * 3 - 5) / b
    elif a == b:
        X = -4
    else:  # a > b
        X = (a**4 + b) / a

    # Вивід результату
    print(f"Результат: X = {X}")
